function truthCheck( collection, pre ) 
{

  let truthyPass = true;

  for(let i in collection)
  {

    if( !(collection[i][pre]) )
    {

      truthyPass = false;

    }


  }


  return truthyPass;


}

truthCheck([{name: "Quincy", role: "Founder", isBot: false}, {name: "Naomi", role: "", isBot: false}, {name: "Camperbot", role: "Bot", isBot: true}], "isBot");
