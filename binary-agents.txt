function binaryAgent(str) 
{

  let strArr = str.split(" ");

  let letterArr = [];

  //parseInt returns ACSI values of the letters which then get translated with string.fromCharCode
  for( let i in strArr )
  {

    letterArr.push( String.fromCharCode(parseInt(strArr[i], 2)) );

  }

  let finalStr = letterArr.join();

  //removes the ","
  finalStr = finalStr.replaceAll(",", "");


  console.log(finalStr);

  return finalStr;


}
