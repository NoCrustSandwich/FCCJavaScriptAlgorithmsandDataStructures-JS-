function orbitalPeriod(arr) {
  const GM = 398600.4418;
  const earthRadius = 6367.4447;
  let T;
  let a;

//let finalArr = [];

for( let i in arr)
{

  a = earthRadius + arr[i].avgAlt;

  

  T = Math.round((2)*(Math.PI)*(Math.sqrt( Math.pow(a,3)/(GM) )));


  arr[i].orbitalPeriod = T;

  delete arr[i].avgAlt;



  //console.log(arr[i]);




}

  console.log(arr);
  return arr;


}
